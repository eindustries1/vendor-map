<script>

class VendorMap extends HTMLElement {

    static get observedAttributes() {

        return ["data"];

    }


    attributeChangedCallback(name, oldValue, newValue) {

        if (name === "data" && newValue) {

            const parsed = JSON.parse(newValue);

            this.initMap(parsed.vendors, parsed.userLocation);

        }

    }


    initMap(vendors, userLocation) {

        const center = vendors.length

            ? { lat: vendors[0].lat, lng: vendors[0].lng }

            : { lat: 18.15, lng: -65.45 };


        const map = new google.maps.Map(document.getElementById("map"), {

            center,

            zoom: 12,

            gestureHandling: "greedy",

            fullscreenControl: false,

            streetViewControl: false,

            mapTypeControl: false

        });


        // USER LOCATION (blue dot)

        if (userLocation) {

            new google.maps.Marker({

                position: userLocation,

                map,

                icon: {

                    path: google.maps.SymbolPath.CIRCLE,

                    scale: 8,

                    fillColor: "#4285F4",

                    fillOpacity: 1,

                    strokeColor: "#ffffff",

                    strokeWeight: 2

                },

                title: "You are here"

            });

        }


        const bounds = new google.maps.LatLngBounds();

        if (userLocation) bounds.extend(userLocation);


        // CUSTOM VENDOR ICON

        const vendorIcon = {

            url: "https://maps.gstatic.com/mapfiles/ms2/micons/red-dot.png",

            scaledSize: new google.maps.Size(32, 32)

        };


        vendors.forEach(v => {

            if (!v.lat || !v.lng) return;


            const marker = new google.maps.Marker({

                position: { lat: v.lat, lng: v.lng },

                map,

                title: v.title,

                icon: vendorIcon

            });


            bounds.extend(marker.position);


            const infoHtml = `

                <div class="info-container">

                    <h3 class="info-title">${v.title}</h3>

                    ${v.image ? `<img class="info-img" src="${v.image}" />` : ""}

                    <p class="info-desc">${v.description || ""}</p>

                    <p class="info-address">${v.addressText || ""}</p>


                    <button id="dirBtn" class="info-btn">Directions</button>

                </div>

            `;


            const infoWindow = new google.maps.InfoWindow({ content: infoHtml });


            marker.addListener("click", () => {

                infoWindow.open(map, marker);


                google.maps.event.addListenerOnce(infoWindow, "domready", () => {

                    const btn = document.getElementById("dirBtn");

                    if (btn) {

                        btn.onclick = () => {

                            const origin = userLocation

                                ? `${userLocation.lat},${userLocation.lng}`

                                : "Current+Location";

                            const dest = `${v.lat},${v.lng}`;

                            window.open(

                                `https://www.google.com/maps/dir/?api=1&origin=${origin}&destination=${dest}`,

                                "_blank"

                            );

                        };

                    }

                });

            });

        });


        map.fitBounds(bounds);

    }

}


customElements.define("vendor-map", VendorMap);
